146     * Recursively filters an array by removing null and empty values.
147     *
148     * Traverses the array structure and removes values that evaluate to false,
149     * with special handling for boolean false and numeric zero which are preserved.
150     * Useful for cleaning the OpenRPC schema by removing optional fields that
151     * weren't set.
152     *
153     * @param  array<mixed, mixed> $array             Array to filter recursively
154     * @param  null|Closure        $callback          Optional callback for custom filtering logic
155     * @param  bool                $removeEmptyArrays Whether to remove arrays that become empty after filtering
156     * @return array<mixed, mixed> Filtered array with empty values removed
157     */
158    private static function arr_filter_recursive(array $array, ?Closure $callback = null, bool $removeEmptyArrays = false): array
159    {
160        foreach ($array as $key => &$value) {
161            if (is_array($value)) {
162                $value = self::arr_filter_recursive($value, $callback, $removeEmptyArrays);
163
164                if ($removeEmptyArrays && !(bool) $value) {
165                    unset($array[$key]);
166                }
167
168                continue;
169            }
170
171            if ($callback instanceof Closure && !$callback($value)) {
172                unset($array[$key]);
173
174                continue;
175            }
176
177            if ((bool) $value) {
178                continue;
179            }
180
181            if (is_bool($value)) {
182                continue;
183            }
184
185            if (is_numeric($value) && $value === 0) {
186                continue;
187            }
188
189            unset($array[$key]);
190        }
191
192        unset($value);

194        return $array;
